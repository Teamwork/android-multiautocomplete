017package com.teamwork.autocomplete.adapter;
018
019import androidx.annotation.MainThread;
020import androidx.annotation.NonNull;
021import androidx.annotation.Nullable;
022
023import com.teamwork.autocomplete.MultiAutoComplete;
024import com.teamwork.autocomplete.filter.TokenFilter;
025import com.teamwork.autocomplete.view.AutoCompleteViewBinder;
026
027import java.util.List;
028
029/**
030 * Interface for a typed adapter for the auto complete view.
031 * <p>
032 * Its purpose is to allow client code to set the data set for the type it represents, and optionally listen to token changes within the
033 * user-typed text with an {@link OnTokensChangedListener}.
034 * <p>
035 * Use the static methods in the {@link Build} class to construct an instance.
036 *
037 * @author Marco Salis
038 */
039public interface AutoCompleteTypeAdapter<Model> {
040
041    /**
042     * Set a list of items for this type adapter.
043     * Note that the list will be processed and indexed asynchronously, and used for the next filtering.
044     * <p>
045     * This method can be called to replace an existing data set, even after the {@link MultiAutoComplete} has been created.
046     * The data set change will be notified and the filter updated accordingly.
047     *
048     * @param items The List of items to set into the type adapter.
049     */
050    @MainThread
051    void setItems(@NonNull List<Model> items);
052
053    /**
054     * Set a {@link OnTokensChangedListener} to listen for changes in the matched tokens in the user-typed text.
055     * <p>
056     * <b>Note:</b> this method can only be called if the token filter returned by {@link TokenFilter#getValidTokenPattern()} is not null,
057     * since it's required to detect the token matches within the user typed string.
058     *
059     * @param listener The listener, or null to reset it.
060     * @throws IllegalStateException when the adapter's {@link TokenFilter#getValidTokenPattern()} is null.
061     * @see OnTokensChangedListener
062     */
063    @MainThread
064    void setOnTokensChangedListener(@Nullable OnTokensChangedListener<Model> listener);
065
066
067    /**
068     * Static factory methods to construct an instance of {@link AutoCompleteTypeAdapter}.
069     */
070    class Build {
071
072        @NonNull
073        public static <M> AutoCompleteTypeAdapter<M> from(@NonNull AutoCompleteViewBinder<M> binder, @NonNull TokenFilter<M> filter) {
074            if (binder == null || filter == null) {
075                throw new IllegalArgumentException("View binder and token filter must not be null");
076            }
077            return new BaseTypeAdapterDelegate<>(binder, filter);
078        }
079    }
080
081}
